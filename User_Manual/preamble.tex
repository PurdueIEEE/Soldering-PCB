\def\@maketitle
{
	\newpage
	\null
	\vskip 1.5em
	\begin{center}
		\includegraphics[width=0.25\textwidth]{ieee_mb_blue_4c}
		\vskip 2.0em
		\let \footnote \thanks
		{
			\LARGE
			\textbf{\@title}
			\par
		}
		\vskip 0.5em
		{
			\large
			\lineskip .5em
			\begin{tabular}[t]{c}
				\@author
			\end{tabular}
			\par
		}
		\vskip 0.5em
		{
			\large
			\@date
		}
	\end{center}
	\par
	\vskip 1.0em
}
